      { id:3, brand:"Mary Kay", name:"Base TimeWise - Bege Médio", price:"R$ 89,90", img:"https://via.placeholder.com/800x600?text=Mary+Kay+Base", desc:"Acabamento natural e uniformizador da pele." },
      { id:4, brand:"Natura", name:"Perfume Feminino 75ml", price:"R$ 129,90", img:"https://via.placeholder.com/800x600?text=Natura+Perfume", desc:"Fragrância marcante com notas florais." },
      { id:5, brand:"Avon", name:"Máscara de Cílios Super Volume", price:"R$ 39,90", img:"https://via.placeholder.com/800x600?text=Avon+Mascara", desc:"Volume instantâneo sem borrar." },
      { id:6, brand:"Mary Kay", name:"Kit Cuidados Noturnos", price:"R$ 149,90", img:"https://via.placeholder.com/800x600?text=Mary+Kay+Kit", desc:"Rotina noturna completa para renovação da pele." }
    ];

    const phoneWhats = "55189997491372";
    const email = "guilherme.eng2024@gmail.com";

    const grid = document.getElementById('grid');
    const qInput = document.getElementById('q');
    const brandSelect = document.getElementById('brand');
    const clearBtn = document.getElementById('clear');
    const noresults = document.getElementById('noresults');
    const modalBack = document.getElementById('modalBack');
    const modalClose = document.getElementById('modalClose');
    const yr = document.getElementById('yr');

    yr.textContent = new Date().getFullYear();

    function renderList(list){
      grid.innerHTML = "";
      if(!list.length){
        noresults.style.display = "block";
        return;
      } else noresults.style.display = "none";

      list.forEach(p=>{
        const card = document.createElement('article');
        card.className = 'card';
        card.innerHTML = `
          <img class="thumb" src="${p.img}" alt="${p.name}">
          <div class="meta">
            <div class="brand-pill small">${p.brand}</div>
            <div class="price">${p.price}</div>
          </div>
          <div>
            <div class="name">${p.name}</div>
            <div class="small" style="margin-top:8px">${p.desc}</div>
          </div>
          <div style="margin-top:auto;display:flex;gap:8px" class="actions">
            <button class="btn btn-outline view" data-id="${p.id}">Ver</button>
            <a class="btn btn-primary" href="https://wa.me/${phoneWhats}?text=${encodeURIComponent('Olá! Tenho interesse no produto: ' + p.name + ' (' + p.brand + ') - ' + p.price)}" target="_blank">Comprar</a>
          </div>
        `;
        grid.appendChild(card);
      });

      // add listeners to view buttons
      Array.from(document.querySelectorAll('.view')).forEach(b=>{
        b.addEventListener('click', e=>{
          const id = parseInt(e.currentTarget.dataset.id,10);
          const prod = PRODUCTS.find(x=>x.id===id);
          if(prod) openModal(prod);
        });
      });
    }

    function filterProducts(){
      const q = qInput.value.trim().toLowerCase();
      const brand = brandSelect.value;
      let list = PRODUCTS.filter(p=>{
        const matchesQ = !q || (p.name + ' ' + p.desc + ' ' + p.brand).toLowerCase().includes(q);
        const matchesB = (brand === 'Todos' || p.brand === brand);
        return matchesQ && matchesB;
      });
      renderList(list);
    }

    function openModal(prod){
      document.getElementById('modalTitle').textContent = prod.name;
      document.getElementById('modalBrand').textContent = prod.brand;
      document.getElementById('modalImg').src = prod.img;
      document.getElementById('modalImg').alt = prod.name;
      document.getElementById('modalName').textContent = prod.name;
      document.getElementById('modalDesc').textContent = prod.desc;
      document.getElementById('modalPrice').textContent = prod.price;
      document.getElementById('whatsLink').href = `https://wa.me/${phoneWhats}?text=${encodeURIComponent('Olá! Tenho interesse no produto: ' + prod.name + ' (' + prod.brand + ') - ' + prod.price)}`;
      document.getElementById('mailLink').href = `mailto:${email}?subject=${encodeURIComponent('Pedido de produto: ' + prod.name)}&body=${encodeURIComponent('Olá,%0D%0AGostaria%20de%20mais%20informações%20sobre%20' + prod.name + '%20(' + prod.brand + ').')}`;
      modalBack.style.display = 'flex';
      modalBack.setAttribute('aria-hidden','false');
    }

    function closeModal(){
      modalBack.style.display = 'none';
      modalBack.setAttribute('aria-hidden','true');
    }

    // events
    qInput.addEventListener('input', filterProducts);
    brandSelect.addEventListener('change', filterProducts);
    clearBtn.addEventListener('click', ()=>{
      qInput.value = '';
      brandSelect.value = 'Todos';
      filterProducts();
    });
    modalClose.addEventListener('click', closeModal);
    modalBack.addEventListener('click', e=>{
      if(e.target === modalBack) closeModal();
    });
    document.addEventListener('keydown', e=>{
      if(e.key === 'Escape') closeModal();
    });

    // init
    renderList(PRODUCTS);
  </script>
